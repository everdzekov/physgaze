For CPU-only systems:

For CUDA 11.8:
pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118
For CUDA 12.1:
pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu121
Installation Commands
Option 1: Install all dependencies at once
pip install -r requirements.txt
Option 2: Install step-by-step (recommended for debugging)

Environment Setup Script
Here's a setup script you can run:
#!/bin/bash
# setup_environment.sh

echo "Setting up PhysGaze environment..."

# Create virtual environment (optional)
python -m venv physgaze_env
source physgaze_env/bin/activate  # On Windows: physgaze_env\Scripts\activate

# Upgrade pip
pip install --upgrade pip

# Install PyTorch with CUDA support (if available)
if [ "$(uname)" == "Darwin" ]; then
    # macOS
    pip install torch torchvision torchaudio
elif command -v nvidia-smi &> /dev/null; then
    # Linux/Windows with NVIDIA GPU
    pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118
else
    # CPU only
    pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cpu
fi

# Install other requirements
pip install numpy scipy matplotlib seaborn tqdm h5py pillow scikit-learn tensorboard

echo "Installation complete!"
echo "To activate the environment: source physgaze_env/bin/activate"
echo "To run PhysGaze: python physgaze.py"

Common Issues and Solutions
Issue 1: CUDA version mismatch
# Check CUDA version
nvidia-smi

# Install matching PyTorch version
# For CUDA 11.8:
pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118
Issue 2: Outdated pip
pip install --upgrade pip setuptools wheel
Issue 3: Memory issues
# In your code, reduce batch size
batch_size = 16  # Instead of 32 or 64
Issue 4: Missing scipy.io
# scipy.io is included in scipy package
# If missing, reinstall scipy
pip install --force-reinstall scipy
Docker Support (Optional)
If you prefer using Docker:
# Dockerfile
FROM pytorch/pytorch:2.0.1-cuda11.7-cudnn8-runtime

WORKDIR /workspace

# Install system dependencies
RUN apt-get update && apt-get install -y \
    wget \
    git \
    libgl1-mesa-glx \
    libglib2.0-0 \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements and install
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy source code
COPY . .

CMD ["python", "physgaze.py"]
